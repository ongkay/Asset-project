# Phase 1 Auth Implementation Plan
> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Deliver the full Phase 1 auth slice so `/login`, `/reset-password`, logout, `app_session`, single-device enforcement, and browser verification all work end-to-end against the real runtime database.

**Architecture:** Keep `src/app` thin. Route-local client components in `(public)` call `next-safe-action` server actions from `src/modules/auth/actions.ts`, while `src/modules/auth/services.ts` orchestrates InsForge Auth, `app_session`, `login_logs`, role redirects, and failed-login state derived from `login_logs` inside the 15-minute window. Reset-password emails should redirect back to `/reset-password` with the email preserved in the `redirectTo` query so the completion flow can sign the user in again with the new password and then create a fresh `app_session`.

**Tech Stack:** Next.js 16 App Router, React 19, TypeScript, next-safe-action v8, react-hook-form v7, zod v4, InsForge SDK, Tailwind CSS v4, shadcn/ui primitives, Playwright

```ts
import "server-only";

import { headers } from "next/headers";

export type AuthRequestMetadata = {
  browser: string | null;
  ipAddress: string;
  os: string | null;
};

function detectBrowser(userAgent: string | null) {
  if (!userAgent) return null;
  if (userAgent.includes("Edg")) return "Edge";
  if (userAgent.includes("Chrome")) return "Chrome";
  if (userAgent.includes("Firefox")) return "Firefox";
  if (userAgent.includes("Safari")) return "Safari";
  return null;
}

function detectOperatingSystem(userAgent: string | null) {
  if (!userAgent) return null;
  if (userAgent.includes("Windows")) return "Windows";
  if (userAgent.includes("Mac OS X")) return "macOS";
  if (userAgent.includes("Linux")) return "Linux";
  if (userAgent.includes("Android")) return "Android";
  if (userAgent.includes("iPhone") || userAgent.includes("iPad")) return "iOS";
  return null;
}

export async function readAuthRequestMetadata(): Promise<AuthRequestMetadata> {
  const headerList = await headers();
  const forwardedFor = headerList.get("x-forwarded-for");
  const userAgent = headerList.get("user-agent");

  return {
    browser: detectBrowser(userAgent),
    ipAddress: forwardedFor?.split(",")[0]?.trim() || "127.0.0.1",
    os: detectOperatingSystem(userAgent),
  };
}
```

```ts
type RecentLoginAttempt = {
  createdAt: string;
  isSuccess: boolean;
};

export async function readProfileByEmail(email: string): Promise<AuthProfile | null> {
  const database = createInsForgeAdminDatabase();
  const { data, error } = await database
    .from("profiles")
    .select("user_id, email, username, public_id, avatar_url, role, is_banned")
    .eq("email", email)
    .maybeSingle<ProfileRow>();

  if (error) throw error;
  return data
    ? {
        avatarUrl: data.avatar_url,
        email: data.email,
        isBanned: data.is_banned,
        publicId: data.public_id,
        role: data.role,
        userId: data.user_id,
        username: data.username,
      }
    : null;
}

export async function readRecentLoginAttemptsByEmail(email: string): Promise<RecentLoginAttempt[]> {
  const database = createInsForgeAdminDatabase();
  const sinceIso = new Date(Date.now() - 15 * 60 * 1000).toISOString();
  const { data, error } = await database
    .from("login_logs")
    .select("is_success, created_at")
    .eq("email", email)
    .gte("created_at", sinceIso)
    .order("created_at", { ascending: false })
    .limit(6);

  if (error) throw error;
  return (data ?? []).map((row) => ({
    createdAt: row.created_at,
    isSuccess: row.is_success,
  }));
}

function resolveRedirectTarget(profile: AuthProfile): AuthRedirectTarget {
  return profile.role === "admin" ? "/admin" : "/console";
}
```

```ts
function buildFailedLoginState(attempts: RecentLoginAttempt[]): FailedLoginState {
  if (attempts.length === 0 || attempts[0]?.isSuccess) {
    return { consecutiveFailures: 0, lastFailedAt: null, shouldOfferReset: false };
  }

  let consecutiveFailures = 0;

  for (const attempt of attempts) {
    if (attempt.isSuccess) break;
    consecutiveFailures += 1;
  }

  return {
    consecutiveFailures,
    lastFailedAt: attempts[0]?.createdAt ?? null,
    shouldOfferReset: consecutiveFailures >= 5,
  };
}

export async function checkLoginEmailStatus(input: LoginEmailInput): Promise<LoginEmailStatusResult> {
  const profile = await readProfileByEmail(input.email);

  if (!profile) {
    return {
      email: input.email,
      failedLoginState: { consecutiveFailures: 0, lastFailedAt: null, shouldOfferReset: false },
      status: "unregistered",
    };
  }

  return {
    email: input.email,
    failedLoginState: buildFailedLoginState(await readRecentLoginAttemptsByEmail(input.email)),
    status: "registered",
  };
}
```

```tsx
"use client";

export function LoginPage({ showDevTools }: { showDevTools: boolean }) {
  const router = useRouter();
  const [loginEmail, setLoginEmail] = useState("");
  const [emailStatus, setEmailStatus] = useState<"idle" | "registered" | "unregistered">("idle");
  const [failedLoginState, setFailedLoginState] = useState<FailedLoginState>({

      setEmailStatus(data.status);

      if (data?.ok) {
        router.replace(data.redirectTo);
        router.refresh();
        return;
      }

      if (data?.failedLoginState) {
        setFailedLoginState(data.failedLoginState);
      }
    },
```

```ts
export async function signUpFromLoginFlow(input: RegisterFromLoginInput): Promise<RegisterFromLoginResult> {
  const metadata = await readAuthRequestMetadata();
  const result = await signUpWithPassword(input);

  if (result.error || !result.data?.user) {
    return {
      ok: false,
      message: "Akun tidak bisa dibuat sekarang. Coba beberapa saat lagi.",
    };
  }

  const profile = await readProfileByUserId(result.data.user.id);

  if (!profile) {
    return {
      ok: false,
      message: "Akun tidak bisa dibuat sekarang. Coba beberapa saat lagi.",
    };
  }

  await createAppSession(profile.userId);
  await writeAuthenticationLog({
    browser: metadata.browser,
    email: profile.email,
    failureReason: null,
    ipAddress: metadata.ipAddress,
    isSuccess: true,
    os: metadata.os,
    userId: profile.userId,
  });

  return {
    ok: true,
    redirectTo: resolveRedirectTarget(profile),
  };
}
```

```ts
      reason: "invalid_credentials",

  const signInResult = await signInWithPassword(input);

  if (signInResult.error || !signInResult.data?.user) {
    await writeAuthenticationLog({
      browser: metadata.browser,

      failureReason: signInResult.error?.message ?? "sign_in_failed",
      ipAddress: metadata.ipAddress,
      isSuccess: false,
```
